Login System

As part of my portfolio project, I developed a login system for the website using the 7shifts API, which is our HR scheduling software. This system allowed us to automatically create employee accounts as they were hired on and deactivate any employees who were no longer with the company. This streamlined the onboarding and offboarding process, and ensured that employee data was always up-to-date.

					
	import wixData from 'wix-data';
	import _ from 'js-sha256';
	
	
	export function loginUser(email, password) {
	  var sha256 = require('js-sha256');
		var token = wixData.query("RegisteredUsers")
			.eq("email",email)
			.eq("password",sha256(password))
			.find()
			.then((results) => {
	
			  if(results.items.length > 0){
				//check to see if there is a login hash.
				if (results.items[0].hash != null && results.items[0].hash != "") {
					return results.items[0].hash;
				}else{
					//users array to update
					var toUpdate = results.items[0];
					
					//The "salt" is added onto the has to randomize it
					var salt = Math.floor(Math.random() * 129700156);
					
					//This updates the users hash and salt
					var hash = sha256(email + salt);
					toUpdate.hash = hash;
					toUpdate.salt = salt;
					toUpdate.tokenCreated = new Date();
					wixData.update("RegisteredUsers", toUpdate)
					  .then((results) => {}).catch((err) => {});
					return hash;
				}
	
			  } else {
				//Return invalid
				return false;
			  }
			})
			.catch((err) => {
			  console.log(err);
			});
		return token
	}
